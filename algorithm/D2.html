# 快速排序

// js实现
var sortArray = function (nums) {
  let len = nums.length;
  if (len <= 1) {
    return nums;
  }
  
//   初始化调用，指针在头尾
    quickSort(nums, 0, len - 1)

    function quickSort(nums, left, right) {
        if(nums.length <= 1) return
        let index = partition(nums, left, right)
        // partition 返回两指针交错时left的位置
        
        // 如果左右还有数据，继续递归调用
        // 由于partition中取了一个中间的数，所以在是否递归时，要将中间数归为右边
        if(left < index - 1) {
         quickSort(nums, left, index - 1)   
        }
        if(right > index ) {
         quickSort(nums,  index,right )   
        }
    }

    return nums
};
var partition = function(nums, left, right) {
            // 选出主元、左右指针
        let pivot = nums[Math.floor((right + left) / 2)], i = left, j=right
        
             // 直到交错前，不断移动指针，直到左边比主元大，右边比主元小，此时交换位置（确保此时指针未交错），继续移动主元，重复  
           
            while (i <= j) {
                while(nums[i] < pivot) {
                    i++;
                }
                while(pivot < nums[j]){
                    j--
                }
                if(i <= j) {
                    swap(nums, i, j)
                    i++
                    j--
                }
            }
            return i
        }
function swap(a, i,j) {
    [ a[i], a[j] ] = [ a[j], a[i] ]
}

# 归并排序

将原始数组分成较小的数字，直到只有一个，接着归并成较大的数组，在归并时排序，直到最后只有一个归并完成的大数组。

var sortArray = function (nums) {
    let len = nums.length;
  if (len <= 1) {
      return nums;
  }
    //   调用函数
  return mergeSortRec(nums)
}
// 递归的执行函数
function mergeSortRec(nums) {
    // 停止条件
    let len = nums.length;
    if(nums.length === 1) {
        return nums
    }
    let mid = Math.floor(len / 2)
    let left = nums.slice(0, mid)
    let right = nums.slice(mid, len)
    // 分成两个数组， 调用合并函数，递归调用执行函数
    // 返回合并的值
    return merge(mergeSortRec(left), mergeSortRec(right))
}
function merge(left, right) {
    // 声明result，迭代左右数组的变量
    let result = [], il = 0, ir = 0
    // 迭代两个数组，直到某一个数组的成员都被排序到result中
    //     迭代时比较左右大小，较小的放到result中
    while(il < left.length && ir < right.length) {
        if(left[il] < right[ir] ) {
            result.push(left[il++])
        } else {
            result.push(right[ir++])
        }
    }
    // 剩下的直接放到result中，因为此时另一个数组已经为空
    while(il < left.length) {
        result.push(left[il++])
    }
    while(ir < right.length) {
        result.push(right[ir++])
    }
    return result 
}
