function insertNode(root, node) {
    if(root.key < node.key) {
        if(root.right) {
            insertNode(root.right, node)
        }else {
            root.right = node
        }
    }else {
        if(root.left) {
            insertNode(root.left, node)
        }else {
            root.left = node
        }
    }
}

traverseFn(node, callback) {

        // 中序
        traverseFn(node.left, callback)
        callback(node.key)
        traverseFn(node.right, callback)

        // 先序
        callback(node.key)
        traverseFn(node.left, callback)
        traverseFn(node.right, callback)

        // 后序
        traverseFn(node.left, callback)
        traverseFn(node.right, callback)
        callback(node.key)

    }
}
